491 
493  d2_36kHz(){
494 
495  // this sets the main clock to 84 MHz
496  hwlib::wait_ms( 1 );
497 
498  // enable the clock to port B
499  PMC->PMC_PCER0 = 1 << ID_PIOB;
500 
501  // disable PIO Control on PB25 and set up for Peripheral B TIOA0
502  PIOB->PIO_PDR = PIO_PB25;
503  PIOB->PIO_ABSR |= PIO_PB25;
504 
505  // enable output on B25
506  PIOB->PIO_OER = PIO_PB25;
507 
508  // enable the clock to the TC0 (peripheral # 27)
509  PMC->PMC_PCER0 = 1 << ID_TC0;
510 
511  // configure TC0 channel 0 for WAVSEL=10 mode
512  // using the main clock / 2 as source
513  // and reset on C match
514  TC0->TC_CHANNEL[ 0 ].TC_CMR =
515  TC_CMR_WAVE
516  | TC_CMR_TCCLKS_TIMER_CLOCK1
517  | TC_CMR_WAVSEL_UP_RC;
518 
519  // configure pin A match clear
520  TC0->TC_CHANNEL[ 0 ].TC_CMR |= TC_CMR_ACPC_SET | TC_CMR_ACPA_CLEAR;
521 
522  // set C match/clear for 36 kHz
523  TC0->TC_CHANNEL[ 0 ].TC_RC = 41'000'000 / 36'000;
524 
525  // set 50% duty cycle
526  TC0->TC_CHANNEL[ 0 ].TC_RA = TC0->TC_CHANNEL[ 0 ].TC_RC / 2;
527 
528  set( 0 );
529 
530  // enable and start
531  TC0->TC_CHANNEL[ 0 ].TC_CCR = TC_CCR_CLKEN | TC_CCR_SWTRG;
532  }
533 
535  //
538  void set( bool b ) override {
539  if( b ){
540  // eable C match -> pin set
541  TC0->TC_CHANNEL[ 0 ].TC_CMR |= TC_CMR_ACPC_SET;
542  } else {
543  // remove pin C match action
544  TC0->TC_CHANNEL[ 0 ].TC_CMR &= ~ ( 0x3 << 18);
545  }
546  }

712 long long int HWLIB_WEAK now_ticks(){
713  static bool init_done = false;
714  if( ! init_done ){
715 
716  // switch to the 84 MHz crystal/PLL clock
717  sam3xa::SystemInit();
718 
719  SysTick->CTRL = 0; // stop the timer
720  SysTick->LOAD = 0xFFFFFF; // use its as a 24-bit timer
721  SysTick->VAL = 0; // clear the timer
722  SysTick->CTRL = 5; // start the timer, 1:1
723 
724  init_done = true;
725  }
726 
727  static unsigned int last_low = 0;
728  static unsigned long long int high = 0;
729 
730  // the timer ticks down, but we want an up counter
731  unsigned int low = 0xFFFFFF - ( SysTick->VAL & 0xFFFFFF );
732  if( low < last_low ){
733 
734  // the timer rolled over, so increment the high part
735  high += 0x1ULL << 24;
736  }
737  last_low = low;
738 
739  // return the aggregated ticks value
740  // the counter runs at 84 MHz
741  return ( low | high );
742 
743 }
