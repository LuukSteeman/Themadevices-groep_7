24 class image_16x16 : public image {
25 private:
26  uint8_t * data;
27 
28  color get_implementation( location pos ) const override {
29  auto x = 16 - pos.x; //wovo !@!
30  return
31  ((
32  data[ ( 2 * pos.y ) + 1 - ( x / 8 ) ] //wovo !@!
33  & ( 0x01 << ( x % 8 ))
34  ) == 0 )
35  ? white
36  : black;
37  }
38 
39 public:
40  constexpr image_16x16():
41  image( location( 16, 16 )),
42  data( nullptr )
43  {}
44 
45  constexpr image_16x16( uint8_t * data ):
46  image( location( 16, 16 )),
47  data( data )
48  {}
49 };

58 public:
59  font_default_16x16(){

61  int c =
62  ( i < 32 ) || ( i == 127 )
63  ? 0
64  : i - 32;
65  images[ i ] = image_16x16( font_16x16_data + 4 + ( c * 32 ));
66  }
67  }
68 
69  const image & operator[]( char c ) const override {
70  return images[ c & 0x7F ];
71  }
