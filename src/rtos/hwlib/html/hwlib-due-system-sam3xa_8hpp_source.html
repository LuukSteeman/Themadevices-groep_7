45 static void SystemInit( void )
46 {
47  /* Set FWS according to SYS_BOARD_MCKR configuration */
48  EFC0->EEFC_FMR = EEFC_FMR_FWS(4);
49  EFC1->EEFC_FMR = EEFC_FMR_FWS(4);
50 
51  /* Initialize main oscillator */
52  if ( !(PMC->CKGR_MOR & CKGR_MOR_MOSCSEL) )
53  {
54  PMC->CKGR_MOR = CKGR_MOR_KEY_PASSWD | SYS_BOARD_OSCOUNT | CKGR_MOR_MOSCRCEN | CKGR_MOR_MOSCXTEN;
55  while ( !(PMC->PMC_SR & PMC_SR_MOSCXTS) )
56  {
57  }
58  }
59 
60  /* Switch to 3-20MHz Xtal oscillator */
61  PMC->CKGR_MOR = CKGR_MOR_KEY_PASSWD | SYS_BOARD_OSCOUNT | CKGR_MOR_MOSCRCEN | CKGR_MOR_MOSCXTEN | CKGR_MOR_MOSCSEL;
62 
63  while ( !(PMC->PMC_SR & PMC_SR_MOSCSELS) )
64  {
65  }
66  PMC->PMC_MCKR = (PMC->PMC_MCKR & ~(uint32_t)PMC_MCKR_CSS_Msk) | PMC_MCKR_CSS_MAIN_CLK;
67  while (!(PMC->PMC_SR & PMC_SR_MCKRDY))
68  {
69  }
70 
71  /* Initialize PLLA */
72  PMC->CKGR_PLLAR = SYS_BOARD_PLLAR;
73  while ( !(PMC->PMC_SR & PMC_SR_LOCKA) )
74  {
75  }
76 
77  /* Switch to main clock */
78  PMC->PMC_MCKR = (SYS_BOARD_MCKR & ~PMC_MCKR_CSS_Msk) | PMC_MCKR_CSS_MAIN_CLK;
79  while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) )
80  {
81  }
82 
83  /* Switch to PLLA */
84  PMC->PMC_MCKR = SYS_BOARD_MCKR;
85  while ( !(PMC->PMC_SR & PMC_SR_MCKRDY) )
86  {
87  }
88 
89  SystemCoreClock = CHIP_FREQ_CPU_MAX;
90 }

92 static void SystemCoreClockUpdate( void )
93 {
94  /* Determine clock frequency according to clock register values */
95  switch ( PMC->PMC_MCKR & PMC_MCKR_CSS_Msk )
96  {
97  case PMC_MCKR_CSS_SLOW_CLK: /* Slow clock */
98  if (SUPC->SUPC_SR & SUPC_SR_OSCSEL)
99  {
100  SystemCoreClock = CHIP_FREQ_XTAL_32K;
101  }
102  else
103  {
104  SystemCoreClock = CHIP_FREQ_SLCK_RC;
105  }
106  break;
107 
108  case PMC_MCKR_CSS_MAIN_CLK: /* Main clock */
109  if (PMC->CKGR_MOR & CKGR_MOR_MOSCSEL)
110  {
111  SystemCoreClock = CHIP_FREQ_XTAL_12M;
112  }
113  else
114  {
115  SystemCoreClock = CHIP_FREQ_MAINCK_RC_4MHZ;
116 
117  switch (PMC->CKGR_MOR & CKGR_MOR_MOSCRCF_Msk)
118  {
119  case CKGR_MOR_MOSCRCF_4_MHz:
120  break;
121 
122  case CKGR_MOR_MOSCRCF_8_MHz:
123  SystemCoreClock *= 2U;
124  break;
125 
126  case CKGR_MOR_MOSCRCF_12_MHz:
127  SystemCoreClock *= 3U;
128  break;
129 
130  default:
131  break;
132  }
133  }
134  break;
135 
136  case PMC_MCKR_CSS_PLLA_CLK: /* PLLA clock */
137  case PMC_MCKR_CSS_UPLL_CLK: /* UPLL clock */
138  if ( PMC->CKGR_MOR & CKGR_MOR_MOSCSEL )
139  {
140  SystemCoreClock = CHIP_FREQ_XTAL_12M;
141  }
142  else
143  {
144  SystemCoreClock = CHIP_FREQ_MAINCK_RC_4MHZ;
145 
146  switch ( PMC->CKGR_MOR & CKGR_MOR_MOSCRCF_Msk )
147  {
148  case CKGR_MOR_MOSCRCF_4_MHz:
149  break;
150 
151  case CKGR_MOR_MOSCRCF_8_MHz:
152  SystemCoreClock *= 2U;
153  break;
154 
155  case CKGR_MOR_MOSCRCF_12_MHz:
156  SystemCoreClock *= 3U;
157  break;
158 
159  default:
160  break;
161  }
162  }
163  if ( (PMC->PMC_MCKR & PMC_MCKR_CSS_Msk) == PMC_MCKR_CSS_PLLA_CLK )
164  {
165  SystemCoreClock *= ((((PMC->CKGR_PLLAR) & CKGR_PLLAR_MULA_Msk) >> CKGR_PLLAR_MULA_Pos) + 1U);
166  SystemCoreClock /= ((((PMC->CKGR_PLLAR) & CKGR_PLLAR_DIVA_Msk) >> CKGR_PLLAR_DIVA_Pos));
167  }
168  else
169  {
170  SystemCoreClock = CHIP_FREQ_UTMIPLL / 2U;
171  }
172  break;
173  }
174 
175  if ( (PMC->PMC_MCKR & PMC_MCKR_PRES_Msk) == PMC_MCKR_PRES_CLK_3 )
176  {
177  SystemCoreClock /= 3U;
178  }
179  else
180  {
181  SystemCoreClock >>= ((PMC->PMC_MCKR & PMC_MCKR_PRES_Msk) >> PMC_MCKR_PRES_Pos);
182  }
183 }

188 static void system_init_flash( uint32_t dw_clk )
189 {
190  /* Set FWS for embedded Flash access according to operating frequency */
191  if ( dw_clk < CHIP_FREQ_FWS_0 )
192  {
193  EFC0->EEFC_FMR = EEFC_FMR_FWS(0);
194  EFC1->EEFC_FMR = EEFC_FMR_FWS(0);
195  }
196  else
197  {
198  if ( dw_clk < CHIP_FREQ_FWS_1 )
199  {
200  EFC0->EEFC_FMR = EEFC_FMR_FWS(1);
201  EFC1->EEFC_FMR = EEFC_FMR_FWS(1);
202  }
203  else
204  {
205  if ( dw_clk < CHIP_FREQ_FWS_2 )
206  {
207  EFC0->EEFC_FMR = EEFC_FMR_FWS(2);
208  EFC1->EEFC_FMR = EEFC_FMR_FWS(2);
209  }
210  else
211  {
212  if ( dw_clk < CHIP_FREQ_FWS_3 )
213  {
214  EFC0->EEFC_FMR = EEFC_FMR_FWS(3);
215  EFC1->EEFC_FMR = EEFC_FMR_FWS(3);
